select 
	({
		[Produto].[Marca].&[Festival de Sabores],
		[Produto].[Marca].&[Pedaços de Frutas],
		[Produto].[Marca].&[Sabor da Montanha]
	}) on rows,
	({
		[Tempo].[Ano].&[2014]
	}) ON COLUMNS
from 
	[COMPLETO]
where
	[Measures].[Faturamento]

	order (

		[Measures].[Faturamento], desc

	[COMPLETO]
